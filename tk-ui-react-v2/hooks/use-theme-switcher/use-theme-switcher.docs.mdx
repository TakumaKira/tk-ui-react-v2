A react hook for managing switching theme, respecting color scheme.

## Get started

Install and use the hook in your components:

```tsx
import { Colors, colorsTokensDark } from '@takumakira-individual/tk-ui-react-v2.themes.colors';
import { useThemeSwitcher, Theme } from '@takumakira-individual/tk-ui-react-v2.hooks.use-theme-switcher';

const ignoreColorSchemeChangeAfterSetThemeEverCalled = false // Default is `true`.

export function MyComponent() {
  const { theme, setTheme } = useThemeSwitcher(ignoreColorSchemeChangeAfterSetThemeEverCalled);
  return (
    <Colors overrides={theme === Theme.DARK ? colorsTokensDark : undefined}>
      <span>The theme is {theme}.</span>
      <button onClick={() => setTheme(theme === Theme.LIGHT ? Theme.DARK : Theme.LIGHT)}>Toggle theme</button>
    </Colors>
  );
}
```
